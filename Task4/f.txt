package main
import "fmt"

func RemoveEven(slice []int) []int {
	result := make([]int, len(slice))
	j := 0
	for i := 0; i < len(slice); i++ {
		if slice[i]%2 != 0{
			result[j] = slice[i]
			j ++
		}
	}
	return result[0 : j]
}

func main(){
	input := []int{0, 3, 2, 5}
	result := RemoveEven(input)
	fmt.Println(result) 
}
